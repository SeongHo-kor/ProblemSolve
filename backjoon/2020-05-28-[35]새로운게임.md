
# 삼성 SW 역량 테스트 기출 문제

## [35]새로운게임2

**코드 동작 방식**
1. 0흰색 이동, 다음 위치에 말 존재하면 위로 얹기
2. 1빨강 이동할때, reverse하고 위로 얹기
3. 2파랑 만나면 반대방향으로 한칸 이동
4. 순서대로 0~K 이동
5. 4개 겹쳐지면 종료
---
```cpp
#include<iostream>
#include<vector>

using namespace std;
struct info
{
	int r, c, d;
};
int N, K;
int area[13][13];
int dr[5] = { 0,0,0,-1,1 };
int dc[5] = { 0,1,-1,0,0 };
vector<int> board[13][13];
vector<info> p;

int find_idx(int r, int c, int idx)
{
	for (int i = 0; i < (int)board[r][c].size(); i++) {
		if (board[r][c][i] == idx)return i;
	}
}
int main()
{
	cin >> N >> K;
	for (int i = 1; i <= N; i++) {
		for (int j = 1; j <= N; j++) {
			cin >> area[i][j];
		}
	}
	for (int i = 0; i < K; i++) {
		info now;
		cin >> now.r >> now.c >> now.d;
		board[now.r][now.c].push_back(i);
		p.push_back(now);
	}
	int T = 0;
	int ans = -1;
	while (1) {
		T++;
		if (T > 1000)break;
		bool isanswer = false;
		for (int i = 0; i < K; i++) {
			int r = p[i].r; int c = p[i].c;
			int d = p[i].d;

			int nr = r + dr[d];
			int nc = c + dc[d];
			int nd = d;

			int idx = -1;
			int cnt = -1;
			int b_size = board[r][c].size();
			if (b_size > 1) {
				idx = find_idx(r, c, i);
				cnt = b_size - idx;
				if (nr<1 || nc<1 || nr>N || nc>N || area[nr][nc] == 2) {
					if (d == 1)nd = 2;
					else if (d == 2)nd = 1;
					else if (d == 3)nd = 4;
					else if (d == 4)nd = 3;
					nr = r + dr[nd];
					nc = c + dc[nd];
					if (nr<1 || nc<1 || nr>N || nc>N || area[nr][nc] == 2) {
						p[i].d = nd;
					}
					else if (area[nr][nc] == 0) {
						p[i].d = nd;
						p[i].r = nr;
						p[i].c = nc;

						for (int i = idx; i < b_size; i++) {
							board[nr][nc].push_back(board[r][c][i]);
							p[board[r][c][i]].r = nr;
							p[board[r][c][i]].c = nc;
						}
						while (cnt--)board[r][c].pop_back();
						if (board[nr][nc].size() >= 4) {
							ans = T;
							isanswer = true;
							break;
						}
					}
					else if (area[nr][nc] == 1) {
						p[i].d = nd;
						p[i].r = nr;
						p[i].c = nc;
						for (int i = b_size - 1; i >= idx; i--) {
							board[nr][nc].push_back(board[r][c][i]);
							p[board[r][c][i]].r = nr;
							p[board[r][c][i]].c = nc;
						}
						while (cnt--)board[r][c].pop_back();
						if (board[nr][nc].size() >= 4) {
							ans = T;
							isanswer = true;
							break;
						}
					}
				}
				else if (area[nr][nc] == 0) {
					p[i].d = nd;
					p[i].r = nr;
					p[i].c = nc;

					for (int i = idx; i < b_size; i++) {
						board[nr][nc].push_back(board[r][c][i]);
						p[board[r][c][i]].r = nr;
						p[board[r][c][i]].c = nc;
					}
					while (cnt--)board[r][c].pop_back();
					if (board[nr][nc].size() >= 4) {
						ans = T;
						isanswer = true;
						break;
					}
					
				}
				else if (area[nr][nc] == 1) {
					p[i].d = nd;
					p[i].r = nr;
					p[i].c = nc;

					for (int i = b_size - 1; i >= idx; i--) {
						board[nr][nc].push_back(board[r][c][i]);
						p[board[r][c][i]].r = nr;
						p[board[r][c][i]].c = nc;
					}
					while (cnt--)board[r][c].pop_back();
					if (board[nr][nc].size() >= 4) {
						ans = T;
						isanswer = true;
						break;
					}
					
				}
			}
			else {
				if (nr<1 || nc<1 || nr>N || nc>N || area[nr][nc] == 2) {
					if (d == 1)nd = 2;
					else if (d == 2)nd = 1;
					else if (d == 3)nd = 4;
					else if (d == 4)nd = 3;
					nr = r + dr[nd];
					nc = c + dc[nd];
					if (nr<1 || nc<1 || nr>N || nc>N || area[nr][nc] == 2) {
						p[i].d = nd;
					}
					else {
						p[i].d = nd;
						p[i].r = nr;
						p[i].c = nc;
						board[nr][nc].push_back(board[r][c][0]);
						board[r][c].pop_back();
						if (board[nr][nc].size() >= 4) {
							ans = T;
							isanswer = true;
							break;
						}
					}
				}
				else {
					p[i].d = nd;
					p[i].r = nr;
					p[i].c = nc;
					board[nr][nc].push_back(board[r][c][0]);
					board[r][c].pop_back();
					if (board[nr][nc].size() >= 4) {
						ans = T;
						isanswer = true;
						break;
					}
				}
			}
		}
		if (isanswer == true)break;
	}

	cout << ans << '\n';
	return 0;
}
```
