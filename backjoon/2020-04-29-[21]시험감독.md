
# 삼성 SW 역량 테스트 기출 문제

## [21] 시험감독


**코드 동작 방식**

1. N개 시험장 Ai~An명 , 총감독 B, 부감독C
2. num = Ai - B 총 감독이 감독 가능한 인원수
 2-1. num값이 0보다 크면 부감독 C로 나누고 나머지 있으면 +1
 2-2. num값이 0보다 작으면 종료


---
```cpp
#include<iostream>
#include<vector>
using namespace std;

vector<int> room;
int N, A, B, C;
long long int cnt = 0;


int C_cnt(int room_num)
{
	int c_cnt = 0;
	if (room[room_num] % C == 0)
	{
		c_cnt = room[room_num] / C;
	}
	else
	{
		c_cnt = (room[room_num] / C);
		c_cnt++;
	}
	return c_cnt;
}
int main()
{
	cin >> N;
	for (int i = 0; i < N; i++)
	{
		cin >> A;
		room.push_back(A);
	}
	cin >> B >> C;
	for (int i = 0; i < N; i++)
	{
		room[i] -= B;
		cnt++;
	}
	for (int i = 0; i < N; i++)
	{
		if (room[i] > 0)
		{
			cnt += C_cnt(i);
		}
	}
	cout << cnt << '\n';
	return 0;
}

```
