```cpp
#include<iostream>

using namespace std;


char Up[10], Down[10], Front[10], Back[10], Left[10], Right[10];
char query[2];

void init() {
	for (int i = 1; i <= 9; i++) {
		Up[i] = 'w';
		Down[i] = 'y';
		Front[i] = 'r';
		Back[i] = 'o';
		Left[i] = 'g';
		Right[i] = 'b';
	}
}
void copy_map(char *A, char *B)
{
	for (int i = 1; i <= 9; i++)
	{
		A[i] = B[i];
	}
}
void rotate(char *A, char dir)
{
	if (dir == '+')//시계
	{
		char temp[10] = { '.',A[7],A[8],A[1],A[2],A[3],A[4],A[5],A[6],A[9] };
		copy_map(A, temp);
	}
	else
	{
		char temp[10] = { '.',A[3],A[4],A[5],A[6],A[7],A[8],A[1],A[2],A[9] };
		copy_map(A, temp);
	}
}
void rotate1(char *A, char dir)
{
	if (dir == '+')//시계
	{
		char temp[10] = { '.',A[5],A[4],A[3],A[2],A[1],A[8],A[7],A[6],A[9] };
		copy_map(A, temp);
	}
	else
	{
		char temp[10] = { '.',A[1],A[8],A[7],A[6],A[5],A[4],A[3],A[2],A[9] };
		copy_map(A, temp);
	}
}
void ans()
{
	cout << Up[1] << Up[2] << Up[3] << '\n';
	cout << Up[8] << Up[9] << Up[4] << '\n';
	cout << Up[7] << Up[6] << Up[5] << '\n';

}
void move(char c, char dir)
{
	char tmp[3] = { 0, };
	if (c == 'F') {
		
		if (dir == '+') {
			rotate(Front, dir);

			tmp[0] = Up[7]; tmp[1] = Up[6]; tmp[2] = Up[5];

			Up[7] = Left[5]; Up[6] = Left[4]; Up[5] = Left[3];
			
			Left[5] = Down[5]; Left[4] = Down[6]; Left[3] = Down[7];
			
			Down[5] = Right[1]; Down[6] = Right[8]; Down[7] = Right[7];
			
			Right[1] = tmp[0]; Right[8] = tmp[1]; Right[7] = tmp[2];
		}
		else
		{
			rotate(Front, dir);
			tmp[0] = Up[5]; tmp[1] = Up[6]; tmp[2] = Up[7];
			Up[5] = Right[7]; Up[6] = Right[8]; Up[7] = Right[1];

			Right[7] = Down[7]; Right[8] = Down[6]; Right[1] = Down[5];

			Down[7] = Left[3]; Down[6] = Left[4]; Down[5] = Left[5];

			Left[3] = tmp[0]; Left[4] = tmp[1]; Left[5] = tmp[2];
		}
	}
	else if (c == 'B') {
		
		if (dir == '+') {
			rotate(Back, dir);
			tmp[0] = Up[1]; tmp[1] = Up[2]; tmp[2] = Up[3];
			Up[1] = Right[3]; Up[2] = Right[4]; Up[3] = Right[5];
			Right[3] = Down[3]; Right[4] = Down[2]; Right[6] = Down[1];
			Down[3] = Left[7]; Down[2] = Left[8]; Down[1]= Left[1];
			Left[7] = tmp[0]; Left[8] = tmp[1]; Left[1] = tmp[2];
		}
		else {
			rotate(Back, dir);
			tmp[0] = Up[1]; tmp[1] = Up[2]; tmp[2] = Up[3];
			Up[1] = Left[7]; Up[2] = Left[8]; Up[3] = Left[1];
			Left[7] = Down[3]; Left[8] = Down[2]; Left[1] = Down[1];
			Down[3] = Right[3]; Down[2] = Right[4]; Down[1] = Right[5];
			Right[3] = tmp[0]; Right[4] = tmp[1]; Right[5] = tmp[2];
		}
	}
	else if (c == 'U') {
		if (dir == '+') {
			rotate(Up, dir);
			tmp[0] = Front[1]; tmp[1] = Front[2]; tmp[2] = Front[3];
			Front[1] = Right[1]; Front[2] = Right[2]; Front[3] = Right[3];
			Right[1] = Back[1]; Right[2] = Back[2]; Right[3] = Back[3];
			Back[1] = Left[1]; Back[2] = Left[2]; Back[3] = Left[3];
			Left[1] = tmp[0];Left[2] = tmp[1];Left[3] = tmp[2];
		}
		else {
			rotate(Up, dir);
			tmp[0] = Front[1]; tmp[1] = Front[2]; tmp[2] = Front[3];
			Front[1] = Left[1]; Front[2] = Left[2]; Front[3] = Left[3];
			Left[1] = Back[1]; Left[2] = Back[2]; Left[3] = Back[3];
			Back[1] = Right[1]; Back[2] = Right[2]; Back[3] = Right[3];
			Right[1] = tmp[0]; Right[2] = tmp[1]; Right[3] = tmp[2];
		}
	}
	else if (c == 'D') {
		if (dir == '+') {
			rotate1(Down, dir);
			tmp[0] = Front[7];tmp[1] = Front[6];tmp[2] = Front[5];
			
			Front[7] = Left[7]; Front[6] = Left[6]; Front[5]=Left[5];
			
			Left[7] = Back[7];Left[6] = Back[6];Left[5] = Back[5];
			
			Back[7] = Right[7]; Back[6] = Right[6]; Back[5] = Right[5];
			
			Right[7] = tmp[0];Right[6] = tmp[1];Right[5] = tmp[2];
		}
		else {
			rotate1(Down, dir);
			tmp[0] = Front[7]; tmp[1] = Front[6]; tmp[2] = Front[5];
			
			Front[7] = Right[7];Front[6] = Right[6];Front[5] = Right[5];
			
			Right[7] = Back[7];Right[6] = Back[6];Right[5] = Back[5];
			
			Back[7] = Left[7];Back[6] = Left[6];Back[5] = Left[5];
			
			Left[7] = tmp[0];Left[6] = tmp[1];Left[5] = tmp[2];
		}
	}
	else if (c == 'L') {
		if (dir == '+') {
			rotate(Left, dir);
			tmp[0] = Up[1];tmp[1] = Up[8];tmp[2] = Up[7];
			
			Up[1] = Back[5]; Up[8] = Back[4]; Up[7] = Back[3];
			
			Back[5] = Down[7]; Back[4] = Down[8]; Back[3] = Down[1];
			
			Down[7] = Front[1]; Down[8] = Front[8]; Down[1] = Front[7];
			
			Front[1] = tmp[0];Front[8] = tmp[1]; Front[7] = tmp[2];
		}
		else {
			rotate(Left, dir);
			tmp[0] = Up[1]; tmp[1] = Up[8]; tmp[2] = Up[7];

			Up[1] = Front[1]; Up[8] = Front[8]; Up[7] = Front[7];

			Front[1] = Down[7];Front[8] = Down[8];Front[7] = Down[1];

			Down[7] = Back[5];Down[8] = Back[4];Down[1] = Back[3];

			Back[5] = tmp[0];Back[4] = tmp[1];Back[3] = tmp[2];

		}
	}
	else if (c == 'R') {
		if (dir == '+') {
			rotate(Right, dir);
			tmp[0] = Up[5];tmp[1] = Up[4];tmp[2] = Up[3];
			
			Up[5] = Front[5];Up[4] = Front[4];Up[3] = Front[3];
			
			Front[5] = Down[3];Front[4] = Down[4];Front[3] = Down[5];
			
			Down[3] = Back[1]; Down[4] = Back[8]; Down[5] = Back[7];
			
			Back[1] = tmp[0];Back[8] = tmp[1];Back[7] = tmp[2];
		}
		else {
			rotate(Right, dir);
			tmp[0] = Up[5]; tmp[1] = Up[4]; tmp[2] = Up[3];
			
			Up[5] = Back[1];Up[4] = Back[8];Up[3] = Back[7];
			
			Back[1] = Down[3];Back[8] = Down[4];Back[7] = Down[5];
			
			Down[3] = Front[5]; Down[4] = Front[4]; Down[5] = Front[3];

			Front[5] = tmp[0];Front[4] = tmp[1];Front[3] = tmp[2];
		}
	}
}
int main()
{
	int tc, n;
	cin >> tc;	for (int i = 0; i < tc; i++) {
		cin >> n;
		init();
		for (int j = 0; j < n; j++) {
			
			cin >> query;
			move(query[0], query[1]);
		}
		ans();
	}
	return 0;
}
```
