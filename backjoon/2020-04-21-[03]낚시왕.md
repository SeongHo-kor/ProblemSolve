# 삼성 SW 역량 테스트 기출 문제

## [03] 낚시왕

https://www.acmicpc.net/problem/17143

**코드 동작 방식**

1. 열(C) 한칸씩 이동하고 행(R) 1~R-1에서 상어 잡기.
2. 상어 이동
   2-1. 상어 한 마리씩 이동하면서 map에 저장
   2-2  map에 이미 상어 있는 경우에는 이동할 상어와 크기 비교
   - 이미 존재하는 상어가 새로운 상어보다 크기가 작으면 새로운 상어 이동x, 반대이면 새로운 상어가 그 자리 차지

---

```cpp
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

struct info {
	int r, c, s, d, z;
};

int R, C, M;
vector<info> shark;
vector<info> map[101][101];
int dr[5] = { 0,-1,1,0,0 };
int dc[5] = { 0,0,0,1,-1 };

int catch_shark(int c_idx)
{
	int size = -1;
	for (int i = 1; i <= R; i++) {
		if ((int)map[i][c_idx].size() == 1) {
			size = map[i][c_idx][0].z;
			map[i][c_idx][0].z = -1;
			map[i][c_idx].pop_back();
			break;
		}
	}
	for (int i = 1; i <= R; i++) {
		for (int j = 1; j <= C; j++) {
			if ((int)map[i][j].size() == 1) {
				shark.push_back(map[i][j][0]);
				map[i][j].pop_back();
			}
		}
	}
	return size;
}
void move_shark() {
	int s_size = shark.size();	
	for (int i = 0; i < s_size; i++) {
		int dir = shark[i].d;
		int speed = shark[i].s;

		shark[i].r = shark[i].r + (speed*dr[dir]) % (R * 2 - 2);
		shark[i].c = shark[i].c + (speed*dc[dir]) % (C * 2 - 2);

		while (shark[i].r < 1 || shark[i].c < 1 || shark[i].r > R || shark[i].c > C){

			if (shark[i].d == 1) {
				shark[i].d = 2;
				shark[i].r = -shark[i].r + 2;
			}
			else if (shark[i].d == 2) {
				shark[i].d = 1;
				shark[i].r = 2 * R - shark[i].r;
			}
			else if (shark[i].d == 3) {
				shark[i].d = 4;
				shark[i].c = 2 * C - shark[i].c;
			}
			else if (shark[i].d == 4) {
				shark[i].d = 3;
				shark[i].c = -shark[i].c + 2;
			}
		}
		
		if (map[shark[i].r][shark[i].c].size() == 1) {
			if (map[shark[i].r][shark[i].c][0].z < shark[i].z) {
				map[shark[i].r][shark[i].c].pop_back();
				map[shark[i].r][shark[i].c].push_back(shark[i]);
			}
		}
		else map[shark[i].r][shark[i].c].push_back(shark[i]);
	}
	shark.clear();
}
int main()
{
	cin >> R >> C >> M;
	for (int i = 0; i < M; i++)
	{
		info now;
		cin >> now.r >> now.c >> now.s >> now.d >> now.z;
		map[now.r][now.c].push_back(now);
	}
	
	int ans = 0;
	for (int i = 1; i <= C; i++) {
		int res = catch_shark(i);
		if (res != -1)ans += res;
		move_shark();
	}

	cout << ans << '\n';

	return 0;
}
```
- 다른 방식의 풀이: 물고기 이동할때, 방향 같고, 현재 위치랑 같게 움직이는 수 R*2-2,C*2-2로 %연산자를 사용해 최대 R,C번만 움직이게 수정
- 기존의 시간복잡도보다 느리긴 하지만 직관적으로 풀고 생각하는데 시간을 단축시킬 수 있었음.  
```cpp
#include<iostream>
#include<vector>
#include<memory.h>
using namespace std;
struct info
{
	int r, c, s, d, z,idx;
};

int R, C, M;
int dr[5] = { 0,-1,1,0,0 };
int dc[5] = { 0,0,0,1,-1 };
vector<info> fish;
int map[101][101];
int temp[101][101];


int fishing_man(int idx)
{
	for (int i = 1; i <= R; i++)
	{
		if (map[i][idx] > 0) {
			int num = fish[map[i][idx] - 1].z;
			fish[map[i][idx]-1].z = -1;
			map[i][idx] = 0;
			return num;
		}
	}
	return 0;
}
void fish_move()
{
	for (int i = 0; i < M; i++){
		if (fish[i].z == -1)continue;

		int r = fish[i].r; int c = fish[i].c;
		int d = fish[i].d; int s = fish[i].s;
		int z = fish[i].z; 

		if (d == 1 || d == 2) {	
			int rotate_R = R * 2 - 2;
			s %= rotate_R;

			for (int k = 1; k <= s; k++) {
				r += dr[d];
				if (r < 1) {
					r = 2;
					d = 2;
				}
				else if (r > R) {
					r = R - 1;
					d = 1;
				}
			}
		}
		else if (d == 3 || d == 4) {
			int rotate_C = C * 2 - 2;
			s %= rotate_C;

			for (int k = 1; k <= s; k++) {
				c += dc[d];
				if (c < 1) {
					c = 2;
					d = 3;
				}
				else if (c > C) {
					c = C - 1;
					d = 4;
				}
			}
		}
		fish[i].r = r; fish[i].c = c;
		if (temp[fish[i].r][fish[i].c] > 0) {
			int f_idx = temp[fish[i].r][fish[i].c] - 1;
			if (fish[i].z < fish[f_idx].z) {
				fish[i].z = -1;
			}
			else {
				fish[f_idx].z = -1;
				temp[fish[i].r][fish[i].c] = fish[i].idx;
			}
		}
		else temp[fish[i].r][fish[i].c] = fish[i].idx;

		fish[i].d = d;
	}
	for (int i = 1; i <= R; i++) {
		for (int j = 1; j <= C; j++) {
			map[i][j] = temp[i][j];
		}
	}
	memset(temp, 0, sizeof(temp));
}
int main()
{
	cin >> R >> C >> M;
	if (M == 0){
		cout << 0 << '\n';
		return 0;
	}
	for (int i = 0; i < M; i++) {
		info now;
		cin >> now.r >> now.c >> now.s >> now.d >> now.z;
		now.idx = i + 1;
		fish.push_back(now);
		map[now.r][now.c] = i + 1;
	}
	int ans = 0;
	for (int i = 1; i <= C; i++) {
		ans+=fishing_man(i);
		fish_move();
	}
	cout << ans << '\n';
	return 0;
}
```
