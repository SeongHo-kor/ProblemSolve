
# 삼성 SW 역량 테스트 기출 문제

## [28]게리맨더링

**코드 동작 방식**

1. 1~N개 중 두개의 선거구 만들기
2. 구역 나누기, 조합 N개중 1개 ~ N개중 N/2개 조합
3. bfs()로 나눈 두개의 선거구가 연결 되어있는지 확인
4. 모두 연결된 경우 두 선거구 인구수 차의 최솟값 구하기

---
```cpp
#include<iostream>
#include<queue>

using namespace std;

int N;
int check[11];
int visited[11][11];
int arr[11];
int ans = 987654321;

int bfs()
{

	int chk1[11] = { 0, };
	int chk2[11] = { 0, };

	queue<int> q1;
	queue<int> q2;

	for (int i = 1; i <= N; i++) {
		if (check[i] == 1) {
			q1.push(i);
			chk1[i] = 1;
			break;
		}
	}
	for (int i = 1; i <= N; i++) {
		if (check[i] == 0) {
			q2.push(i);
			chk2[i] = 1;
			break;
		}
	}
	while (!q1.empty()) {
		int cur = q1.front();
		q1.pop();
		for (int i = 1; i <= N; i++) {
			if (i == cur)continue;
			if (chk1[i] == 1)continue;
			if (visited[cur][i] == 1) {
				if (check[i] == 1) {
					chk1[i] = 1;
					q1.push(i);
				}
			}
		}
	}
	int res1 = 0;
	for (int i = 1; i <= N; i++) {
		if (chk1[i] == 1 && check[i] == 1)res1 += arr[i];
		else if (chk1[i] == 0 && check[i] == 1)return -1;
	}

	while (!q2.empty()) {
		int cur = q2.front();
		q2.pop();
		for (int i = 1; i <= N; i++) {
			if (i == cur)continue;
			if (chk2[i] == 1)continue;
			if (visited[cur][i] == 1) {
				if (check[i] == 0) {
					chk2[i] = 1;
					q2.push(i);
				}
			}
		}
	}
	int res2 = 0;
	for (int i = 1; i <= N; i++) {
		if (chk2[i] == 1 && check[i] == 0)res2 += arr[i];
		else if (chk2[i] == 0 && check[i] == 0)return -1;
	}
	return abs(res1 - res2);
}
void combi(int cnt, int idx, int num)
{
	if (cnt == num) {
		int res = bfs();
		if (res == -1)return;
		else {
			if (ans > res) ans = res;
		}
		return;
	}
	for (int i = idx; i <= N; i++) {
		if (check[i] == 1)continue;
		check[i] = 1;
		combi(cnt + 1, i, num);
		check[i] = 0;
	}
}


int main()
{
	cin >> N;
	for (int i = 1; i <= N; i++) {
		cin >> arr[i];
	}
	for (int i = 1; i <= N; i++) {
		int t;
		cin>>t;
		for (int j = 0; j < t; j++) {
			int n;
			cin >> n;
			visited[i][n] = 1;
		}
	}

	int t_cnt = N / 2;
	if (N % 2 != 0)t_cnt++;

	for (int i = 1; i <= t_cnt; i++) {
		combi(0, 1, i);
	}
	if (ans == 987654321)cout << -1 << '\n';
	else cout << ans << '\n';
	return 0;
}
```
