
# 삼성 SW 역량 테스트 기출 문제(SWEA)

## [13]벌꿀채취

**코드 동작 방식**


---
```cpp
#include<iostream>
#include<vector>
#include<algorithm>
#include<memory.h>

using namespace std;

int N, M, C;
int area[11][11];
int visited[11][11];
int check[6];
vector<int> v[2];
int answer = 0;
int ans[2];
void choose_honey(int idx, int cnt, int v_idx, int F)
{
	if (cnt == F) {

		int res = 0;
		for (int i = 0; i < M; i++) {
			if (check[i] == 1) {
				res += v[v_idx][i];
			}
		}
		if (res <= C) {
			int res1 = 0;
			for (int i = 0; i < M; i++) {
				if (check[i] == 1) {
					res1 += (v[v_idx][i])*(v[v_idx][i]);
				}
			}
			if (ans[v_idx] < res1)ans[v_idx] = res1;
		}
		return;
	}
	for (int i = idx; i < M; i++) {
		if (check[i] == 1)continue;
		check[i] = 1;
		choose_honey(i, cnt + 1, v_idx, F);
		check[i] = 0;
	}
}
void choose_honeyBowl(int c_idx, int r_idx, int cnt)
{
	if (cnt == 2) {
		memset(ans, 0, sizeof(ans));
		for (int i = 1; i <= M; i++) {
			choose_honey(0, 0, 0, i);
			choose_honey(0, 0, 1, i);
		}
		if (answer < ans[0] + ans[1])answer = ans[0] + ans[1];
		return;
	}
	for (int i = 0; i < N; i++) {
		for (int j = r_idx; j < N; j++) {
			if (visited[i][j] == 1)continue;
			if (j + (M - 1) >= N)break;
			for (int k = 0; k < M; k++) {
				visited[i][j + k] = 1;
				v[cnt].push_back(area[i][j + k]);
			}
			choose_honeyBowl(i, j, cnt + 1);
			for (int k = 0; k < M; k++) {
				visited[i][j + k] = 0;
				v[cnt].pop_back();
			}
		}
	}
}

int main()
{
	int T;
	cin >> T;
	for (int tc = 1; tc <= T; tc++) {
		cin >> N >> M >> C;
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				cin >> area[i][j];
			}
		}
		choose_honeyBowl(0, 0, 0);
		cout << '#' << tc << ' ' << answer << '\n';
		memset(area, 0, sizeof(area));
		memset(ans, 0, sizeof(ans));
		answer = 0;

	}
	return 0;
}
```

**한 번에 구현하지 못했던 이유(실수)**

- 시간 초과, bfs를 크기 1~21까지 하려고 해 시간초과가 남, 왜냐하면 21크기의 bfs를 하면 그 과정에 1,2...의 크기도 구해지기 때문에 중복해서 할 필요가 없다는 생각을 하지 못함
