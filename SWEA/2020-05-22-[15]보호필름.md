```cpp
#include<iostream>
#include<memory.h>

using namespace std;


int area[13][21];
int D, W, K;
bool check = false;
int ans = 21;
void copy_map(int A[21], int B[21]) {
	for (int i = 0; i < W; i++) {
		A[i] = B[i];
	}
}
bool ispass()
{
	int res = 0;
	for (int i = 0; i < W; i++) {
		int B = area[0][i];
		int cnt = 1;
		for (int j = 1; j < D; j++) {
			if (B == area[j][i])cnt++;
			else {
				B = area[j][i];
				cnt = 1;
			}
			if (cnt == K)break;
		}
		if (cnt == K)continue;
		else return false;
	}
	return true;
}

void fill_m(int idx, int n)
{
	for (int i = 0; i < W; i++) {
		area[idx][i] = n;
	}
}
void combi(int cnt, int idx) {
	if (ans < cnt)return;
	if (idx >= D)
	{
		if (ispass()) {
			if (ans > cnt)ans = cnt;
		}
		return;
	}
	int temp[21];
	combi(cnt, idx + 1);

	copy_map(temp, area[idx]);
	fill_m(idx, 0);
	combi(cnt + 1, idx + 1);


	fill_m(idx, 1);
	combi(cnt + 1, idx + 1);
	copy_map(area[idx], temp);

}
int main()
{
	int T;
	cin >> T;
	for (int tc = 1; tc <= T; tc++) {
		cin >> D >> W >> K;
		for (int i = 0; i < D; i++) {
			for (int j = 0; j < W; j++) {
				cin >> area[i][j];
			}
		}

		if (ispass() == true) {
			cout << '#' << tc << ' ' << 0 << '\n';
		}
		else {
			combi(0, 0);
			cout << '#' << tc << ' ' << ans << '\n';
		}
		ans = 21;
		memset(area, 0, sizeof(area));
	}
	return 0;
}
```
